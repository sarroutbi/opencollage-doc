\section{Communication Protocols}

\begin{frame}{Standardized Agent Communication}
  \vspace{-0.3cm}
  \scriptsize
  \begin{block}{Core Message}
    \scriptsize
    OpenCollage defines a typed, event-driven communication protocol with a
    schema registry, enabling agents from different frameworks to exchange
    structured messages without custom adapters.
  \end{block}

  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{\faEnvelope\ \ Message Contract:}
      \begin{itemize}
        \item Typed payloads with JSON Schema validation
        \item Versioned schemas in a central registry
        \item Backward-compatible evolution rules
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{\faBolt\ \ Transport Modes:}
      \begin{itemize}
        \item Synchronous request/response
        \item Async event-driven (pub/sub)
        \item Streaming for long-running agents
      \end{itemize}
    \end{column}
  \end{columns}
