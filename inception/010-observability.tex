\section{Observability}

\begin{frame}{Native Observability with OpenTelemetry}
  \begin{block}{Core Message}
    \fontsize{6}{7}\selectfont
    Every agent invocation, handoff, and state transition emits structured
    telemetry---traces, metrics, and logs---through the OpenTelemetry standard,
    enabling end-to-end pipeline visibility.
  \end{block}

  \vspace{-0.1cm}

  \vspace{-0.2cm}
  \scriptsize
  \begin{itemize}
    \item \textbf{Distributed traces}: Follow a task from user prompt through every agent in the DAG
    \item \textbf{Token \& cost metrics}: Per-agent, per-pipeline LLM cost attribution
    \item \textbf{Latency \& error budgets}: SLO-driven alerting on pipeline performance
    \item \textbf{Vendor-neutral}: Export to Jaeger, Datadog, Splunk, or any OTLP-compatible backend
  \end{itemize}
\end{frame}
