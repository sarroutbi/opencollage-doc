\section{Agent Registry \& Packaging}

\begin{frame}{\fontsize{10}{12}\selectfont Agent Registry: The Container Registry Moment for AI Agents}
  \vspace{-0.2cm}
  \begin{block}{\scriptsize Core Message}
    \scriptsize
    Containers became ubiquitous when registries (Docker Hub, Quay.io) gave them
    a standard packaging, versioning, and distribution model. AI agents need the
    same infrastructure---an \textbf{Agent Registry} that makes agents discoverable,
    versionable, and deployable with a single command.
  \end{block}

\begin{frame}[fragile]{\fontsize{10}{12}\selectfont Agent Manifest: The Dockerfile for AI Agents}
  \vspace{-0.1cm}
  \begin{block}{\scriptsize Core Message}
    \scriptsize
    Every agent is packaged with a declarative manifest that describes its identity,
    capabilities, dependencies, and trust metadata---making agents reproducible,
    auditable, and portable across environments.
  \end{block}

  \vspace{-0.5cm}

  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \begin{lstlisting}[language=,basicstyle=\ttfamily\fontsize{4}{4.8}\selectfont,title={\tiny\bfseries agent-manifest.yaml}]
apiVersion: opencollage/v1
kind: AgentPackage
metadata:
  name: security-analyst
  version: 2.1.0
  author: acme-corp
  license: Apache-2.0
  registry: registry.opencollage.io

spec:
  framework: crewai
  adapter: crewai-adapter:1.4.0

  model:
    provider: openai
    name: gpt-4o
    version: "2026-02-01"
    temperature: 0.1

  tools:
    - name: siem_query
      version: ">=1.0.0"
    - name: asset_lookup
      version: ">=2.3.0"

  role: "Senior SOC Analyst"
  system_prompt_ref: prompts/soc-analyst.md

  resources:
    max_tokens: 4096
    timeout: 120s
    max_tool_calls: 20
      \end{lstlisting}
    \end{column}
    \begin{column}{0.50\textwidth}
      \begin{lstlisting}[language=,basicstyle=\ttfamily\fontsize{4}{4.8}\selectfont,title={\tiny\bfseries agent-manifest.yaml (cont.)}]
  dependencies:
    python: ">=3.11"
    packages:
      - crewai>=0.30.0
      - langchain-openai>=0.1.0
    agents:  # compose with other agents
      - registry.opencollage.io/
          threat-intel:1.0.0
      - registry.opencollage.io/
          vuln-scanner:2.0.0

  conformance:
    passed: true
    suite_version: 1.2.0
    badge: gold

  provenance:
    signed_by: acme-corp-signing-key
    signature_format: sigstore
    sbom_format: spdx
    attestations:
      - type: build
        builder: github-actions
      - type: conformance
        verifier: o9e-conformance-ci
      \end{lstlisting}

      \vspace{0.1cm}

        \item \texttt{o9ectl push security-analyst:2.1.0}
        \item \texttt{o9ectl pull security-analyst:2.1.0}
      \end{enumerate}
