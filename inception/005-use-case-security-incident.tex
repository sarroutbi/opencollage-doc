\section{Use Case: Security Incident Response}

\begin{frame}{\fontsize{10}{12}\selectfont Real-World Use Case: Automated Security Incident Response}
  \fontsize{6}{7}\selectfont
  \begin{alertblock}{The Scenario}
    \fontsize{6}{7}\selectfont
    A SOC team needs an automated pipeline that triages security alerts, correlates
    threat intelligence, generates remediation code, and produces executive reports---each
    step demanding a \textbf{different agent specialization} from a \textbf{different framework}.
  \end{alertblock}

  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Why this needs OpenCollage:}
      \begin{itemize}
        \item CrewAI excels at role-based analysis and writing
        \item AutoGen excels at code generation with tool use
        \item LangGraph excels at stateful graph-based reasoning
        \item No single framework covers all requirements
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Pipeline characteristics:}
      \begin{itemize}
        \item Conditional branching on alert severity
        \item Parallel fan-out for threat intel + vuln analysis
        \item Fan-in synchronization before code gen
        \item End-to-end tracing across 4 frameworks
      \end{itemize}
    \end{column}
  \end{columns}
