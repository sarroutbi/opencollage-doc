\begin{frame}[fragile]{\fontsize{11}{13}\selectfont Security Incident Pipeline: Full YAML Specification (2/2)}

    \begin{column}{0.50\textwidth}
      \begin{lstlisting}[language=,basicstyle=\ttfamily\fontsize{4}{4.8}\selectfont,title={\tiny\bfseries incident-response.yaml --- DAG \& Routing}]
  dag:
    edges:
      - from: alert-triage
        to: [threat-intel, vuln-analysis]
        condition:
          field: severity
          operator: in
          values: [HIGH, CRITICAL]
        mode: parallel

      - from: alert-triage
        to: vuln-analysis
        condition:
          field: severity
          operator: in
          values: [MEDIUM, LOW]

      - from: [threat-intel, vuln-analysis]
        to: remediation-codegen
        join: all  # fan-in: wait for both

      - from: remediation-codegen
        to: human-approval

      - from: human-approval
        to: executive-report
        condition:
          field: decision
          operator: eq
          values: [approved]
      \end{lstlisting}
    \end{column}
    \begin{column}{0.50\textwidth}
      \begin{lstlisting}[language=,basicstyle=\ttfamily\fontsize{4}{4.8}\selectfont,title={\tiny\bfseries incident-response.yaml --- Handoff Schemas}]
  schemas:
    TriageResult:
      type: object
      required: [alert_id, severity, summary]
      properties:
        alert_id:
          type: string
          format: uuid
        severity:
          type: string
          enum: [LOW, MEDIUM, HIGH, CRITICAL]
        summary:
          type: string
          maxLength: 2048
        affected_assets:
          type: array
          items:
            type: object
            properties:
              hostname: {type: string}
              ip: {type: string, format: ipv4}
              criticality: {type: string}

  handoffs:
    - from: alert-triage
      to: [threat-intel, vuln-analysis]
      schema: TriageResult
    - from: [threat-intel, vuln-analysis]
      to: remediation-codegen
      schema: EnrichedIncident
    - from: remediation-codegen
      to: human-approval
      schema: RemediationPlan
    - from: human-approval
      to: executive-report
      schema: ApprovedRemediation
      \end{lstlisting}
    \end{column}
  \end{columns}
