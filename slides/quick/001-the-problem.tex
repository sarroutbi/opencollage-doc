\section{The Problem}

\begin{frame}{AI Agents Are Powerful---But Isolated}
  \scriptsize
  \begin{alertblock}{Core Message}
    \scriptsize
    AI agent frameworks are incompatible. Agents built in one framework
    cannot collaborate with agents in another---creating silos, vendor lock-in,
    and duplicated effort.
  \end{alertblock}
